# 알고리즘 공부
## 올림 처리: 피자 나눠 먹기 (3)

```sql
-- 방법1
SELECT 
    PT_NAME, 
    PT_NO, 
    GEND_CD, 
    AGE, 
    CASE 
        WHEN TLNO IS NULL THEN 'NONE' -- 전화번호가 NULL이면 'NONE' 출력
        ELSE TLNO                   -- 그렇지 않으면 기존 전화번호 출력
    END AS TLNO
FROM PATIENT
WHERE AGE <= 12 AND GEND_CD = 'W'
ORDER BY AGE DESC, PT_NAME ASC;
```

```sql
-- 방법2
SELECT  
    PT_NAME, 
    PT_NO, 
    GEND_CD, 
    AGE,
    COALESCE(TLNO, 'NONE')
FROM PATIENT
WHERE AGE <= 12 AND GEND_CD = 'W'
ORDER BY AGE DESC, PT_NAME ASC;
```
